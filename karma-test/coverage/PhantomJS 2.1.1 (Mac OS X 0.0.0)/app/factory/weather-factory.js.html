(function () {
	var fM = angular.module('karma.func.factory');
 
	fM.factory('getWeatherCoopreative', ['$http', '$q', function($http, $q) {
		var getWeatherFactory = {};
 
		getWeatherFactory.grabWeather = function (city) {
			var _def = $q.defer();
			var apiKey = 'd6882824ad1df12b290efaa9ab18cc56';
			var apiLane = 'http://api.openweathermap.org/data/2.5/weather?q='+city+'&appid='+apiKey;
			$http.get(apiLane)
			.success(function (responseWeatherPackage) {
				var wp = responseWeatherPackage;
				console.log('12 -- wp: ', wp);
				return _def.resolve(wp);
			})
			.error(function(msg, config, status) {
				var weatherErrorObj = {
					success: false,
					message: msg,
					cfg: config,
					status: status
				};
				return _def.reject(weatherErrorObj);
			});
			
			return _def.promise;
		};
 
		return getWeatherFactory;
	}]);
 
})();
